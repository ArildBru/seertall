import { BlobServiceClient } from "@azure/storage-blob"; import * as XLSX from "xlsx"; export default async function handler(req, res) { try { if (req.method !== "POST") { return res.status(405).json({ ok: false, error: "Method not allowed" }); } const { uke } = req.body; const file = req.body.file; // base64 fra frontend if (!uke || !file) { return res.status(400).json({ ok: false, error: "Missing week or file" }); } // ------------------------------------------------------- // 1. Les Excel // ------------------------------------------------------- const buffer = Buffer.from(file, "base64"); const workbook = XLSX.read(buffer, { type: "buffer" }); const sheet = workbook.Sheets[workbook.SheetNames[0]]; const rows = XLSX.utils.sheet_to_json(sheet); if (!rows || rows.length === 0) { return res.status(400).json({ ok: false, error: "Empty Excel file" }); } // ------------------------------------------------------- // 2. Finn kanalnavn fra Excel // ------------------------------------------------------- const kanalRaw = rows[0].Mediehus || rows[0].mediehus || rows[0].Kanal || rows[0].kanal; if (!kanalRaw) { return res.status(400).json({ ok: false, error: "Could not detect channel in Excel" }); } // Normaliser kanalnavn const kanal = kanalRaw.toLowerCase().replace(/\s+/g, ""); // ------------------------------------------------------- // 3. Konverter Excel-rader til vårt JSON-format // ------------------------------------------------------- const jsonRows = rows.map((r) => ({ program: r.Tittel || r.tittel || "", episode: Number(r.Episode || r.episode || 0), uke: Number(uke), lineart: Number(r.Lineært || r["Lineært"] || r.Lineart || r.lineart || 0), vod: Number(r.VOD || r.vod || 0), total: Number(r.Totalt || r.totalt || 0), kanal: kanalRaw })); // ------------------------------------------------------- // 4. Lag filnavn som aldri overskriver // ------------------------------------------------------- const filename = `uke-${uke}-${kanal}.json`; // ------------------------------------------------------- // 5. Lagre i Azure Blob Storage // ------------------------------------------------------- const blobService = BlobServiceClient.fromConnectionString(process.env.AZURE_STORAGE_CONNECTION_STRING); const container = blobService.getContainerClient("weeks"); const blob = container.getBlockBlobClient(filename); await blob.upload(JSON.stringify({ ok: true, filename, rows: jsonRows }, null, 2), Buffer.byteLength(JSON.stringify(jsonRows))); // ------------------------------------------------------- // 6. Ferdig // ------------------------------------------------------- return res.status(200).json({ ok: true, filename, uke, kanal: kanalRaw, rader: jsonRows.length }); } catch (err) { console.error("Import error:", err); return res.status(500).json({ ok: false, error: err.message }); } }
